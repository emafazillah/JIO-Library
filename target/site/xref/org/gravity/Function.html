460 static private void ZipFolder(String path, String srcFolder, ZipOutputStream zip)throws Exception {
461 	File folder = new File(srcFolder);
462 
463 	for (String fileName : folder.list()) {
464       if (path.equals("")) {
465     	  ZipFile(folder.getName(), srcFolder + "/" + fileName, zip);

472 static private void ZipFile(String path, String srcFile, ZipOutputStream zip)throws Exception {
473 
474     File folder = new File(srcFile);
475     if (folder.isDirectory()) {
476     	ZipFolder(path, srcFile, zip);
477     } 
478     else{
479       byte[] buf = new byte[1024];
480       int len;
481       FileInputStream in = new FileInputStream(srcFile);
482       zip.putNextEntry(new ZipEntry(path + "/" + folder.getName()));
483       while ((len = in.read(buf)) > 0) {
484         zip.write(buf, 0, len);
485       }
486     }
487   } 
488 /**
489  * Function name: unzipFolder
490  * 
491  * <p>
492  * Pointed folder will compressed into zip file and store in the same directory.
493  * <p>
494  * @param srcDir Source directory of zipped folder.
495  * @param destDir Destination directory to be extract.
496  * <p>
497  * Example:
498  * <p>
499  * Function.unzipFolder("C://test/test1.zip", "C://test/");
500  */   
501 public static void unzipFolder( String srcDir, String destDir ) {
502 	try {
503 		ZipFile zipfile = new ZipFile(srcDir);
504 		Enumeration< ? extends ZipEntry> zipEnum = zipfile.entries();
505 		
506 			while( zipEnum.hasMoreElements() ) {
507 				ZipEntry zipentry = (ZipEntry) zipEnum.nextElement();
508 
509 				if (zipentry.isDirectory()) {
510 					File newdir = new File(destDir + File.separator + zipentry.getName());
511 					newdir.mkdir();
512 				} 
513 				else{
514 					String extractPath = destDir + File.separator + zipentry.getName();
515 					File extractFile = new File(extractPath);
516 					if (!extractFile.getParentFile().exists()) {
517 						extractFile.getParentFile().mkdirs();
518 					}else{
519 						System.out.println("Info: Folder and data "+'"'+zipentry.getName()+'"'+" already existed at extract location.");
520 						System.out.println("Info: Replace existing content with new extract data.");
521 					}
522         
523 					InputStream input = zipfile.getInputStream(zipentry);
524 					FileOutputStream output = new FileOutputStream(extractPath);
525 					int x;
526 					while( (x = input.read()) != -1 ) {
527 						output.write(x);
528 					}
529 					input.close();
530 					output.close();
531 				}
532 			}
533 			zipfile.close();
534 			System.out.println("Executed: Folder "+srcDir+" successfully extracted.");
535 			System.out.println("Extract location: "+destDir);
536 			} catch (Exception e) {
537 				e.printStackTrace();
538 			}
539 }
